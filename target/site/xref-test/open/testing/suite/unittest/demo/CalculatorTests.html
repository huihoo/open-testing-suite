10  class CalculatorTests {
11  
12      @Test
13      @DisplayName("1 + 1 = 2")
14      void addsTwoNumbers() {
15          Calculator calculator = new Calculator();
16          assertEquals(2, calculator.add(1, 1), "1 + 1 should equal 2");
17      }
18  
19      @ParameterizedTest(name = "{0} + {1} = {2}")
20      @DisplayName("2 + 2 = 4")
21      @CsvSource({
22              "0,    1,   1",
23              "1,    2,   3",
24              "49,  51, 100",
25              "1,  100, 101",
26              "100, 100, 200"
27      })
28      void add(int first, int second, int expectedResult) {
29          Calculator calculator = new Calculator();
30          assertEquals(expectedResult, calculator.add(first, second),
31                  () -> first + " + " + second + " should equal " + expectedResult);
32      }
33  }

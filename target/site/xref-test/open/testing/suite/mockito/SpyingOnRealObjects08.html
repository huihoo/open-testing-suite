18  
19          // optionally, you can stub out some methods:
20          when(spy.size()).thenReturn(100);
21  
22          // using the spy calls *real* methods
23          spy.add("one");
24          spy.add("two");
25  
26          // prints "one" - the first element of a list
27          System.out.println(spy.get(0));
28          System.out.println(spy.get(1));
29  
30          // size() method was stubbed - 100 is printed
31          int spySize =spy.size();
32          System.out.println("Spy size"+spySize);
33          int listSize = list.size();
34          System.out.println("list size"+listSize);
35  
36          assertEquals(100,spySize);
37          assertEquals(0,listSize);
38  
39          assertEquals(spy.get(0),"one");
40          //assertEquals(list.get(0),"one");
41  
42          // optionally, you can verify
43          verify(spy).add("one");
44          verify(spy).add("two");

11  public class SecondStubbing02 {
12      @Test
13      public void test() {
14          // You can mock concrete classes, not only interfaces
15          @SuppressWarnings("unchecked")
16          LinkedList<String> mockedList =(LinkedList<String>) mock(LinkedList.class);
17  
18          // stubbinging
19          when(mockedList.get(0)).thenReturn("first");
20          when(mockedList.get(1)).thenThrow(new RuntimeException());
21          when(mockedList.get(998)).thenReturn("999");
22  
23          // following prints "first"
24          System.out.println(mockedList.get(0));
25  
26          // following throws runtime exception
27          try{
28              System.out.println(mockedList.get(1));
29              fail("Exception was not throw");
30          }catch (Exception e){
31  
32          }
33          // following prints "null" because get(999) was not stubbed
34          System.out.println(mockedList.get(999));
35  
36          verify(mockedList).get(0);
37  
38      }

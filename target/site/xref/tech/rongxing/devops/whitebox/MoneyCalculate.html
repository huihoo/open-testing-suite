7        * 题目四——计算利润的提成
8        * @param money 利润金额
9        * @return 返回double类型的提成
10       */
11      public double getMoney(double money){
12          double sum = 0;
13          if(money<=100000)
14          {
15              sum = 0.1 * money;
16          }
17          if(money>100000 && money<=200000)
18          {
19              sum = 10000 + (money - 100000) * 0.075;
20          }
21          if(money>200000 && money<=400000)
22          {
23              sum = 10000 + 100000 * 0.075 + (money - 200000)*0.05;
24          }
25          if(money>400000 && money<=600000)
26          {
27              sum = 10000 + 100000 * 0.075 + 200000 * 0.05 + (money - 400000) * 0.03;
28          }
29          if(money>600000 && money<=1000000)
30          {
31              sum = 10000 + 100000 * 0.075 + 200000 * 0.05 + 200000 * 0.03 + (money - 600000) * 0.015;
32          }
33          if(money>1000000)
34          {
35              sum = 10000 + 100000 * 0.075 + 200000 * 0.05 + 200000 * 0.03 + 400000 * 0.015 + (money - 1000000) * 0.01;
36          }
37          return sum;
38      }
